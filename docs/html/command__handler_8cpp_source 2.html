54estts::Status command_handler::execute_str(estts::waiting_command *command) {
55 try {
56 SPDLOG_INFO("Sending command");
57 bool retry = true;
58 int retries = 0;
59 // Try to transmit frame
60 while (ti->transmit(command->frame) != estts::ES_OK) {
61 spdlog::error("Failed to transmit frame. Waiting {} seconds", estts::ESTTS_RETRY_WAIT_SEC);
62 sleep_until(system_clock::now() + seconds(estts::ESTTS_RETRY_WAIT_SEC));
63 retries++;
64 if (retries > estts::endurosat::MAX_RETRIES) return estts::ES_UNSUCCESSFUL;
65 SPDLOG_INFO("Retrying transmit ({}/{})", retries, estts::ESTTS_MAX_RETRIES);
66 }
67 // If we got this far,
68 SPDLOG_DEBUG("Successfully transmitted command");
69 }
70 catch (const std::exception &e) {
71 // TODO catch exceptions & do something smart with them
72 spdlog::error("We failed somewhere");
73 return estts::ES_UNSUCCESSFUL;
74 }
75
76 SPDLOG_INFO("Waiting for a response from EagleSat II");
77 sleep_until(system_clock::now() + milliseconds (100));
78 auto telem = ti->receive();
79 if (telem.empty())
80 return estts::ES_UNSUCCESSFUL;
81 SPDLOG_DEBUG("Got response from EagleSat II");
82
83 // todo there are likely error cases here that aren't accounted for. Find & fix them
84
85 if (command->str_callback != nullptr)
86 if (estts::ES_OK != command->str_callback(telem))
87 return estts::ES_UNSUCCESSFUL;
88
89 auto temp_completed = new completed;
90 temp_completed->serial_number = command->serial_number;
91 temp_completed->response_code = estts::ES_OK;
92 completed_cache.push_back(temp_completed);
93
94 return estts::ES_OK;
95}

104estts::Status command_handler::map_telemetry_to_dispatched(const std::vector<estts::telemetry_object *> &telem) {
105 // This function requires some intelligence. The basic premise is that the decoded telemetry objects
106 // need to be mapped back to their associated dispatched command objects. There are a couple of ways we
107 // can handle this. For this version, we're just going to handle each response individually and in real-time.
108
109 while (!dispatched.empty()) {
110 auto current = dispatched.back();
111 auto command_id = current->command->commandID;
112 auto command_address = current->command->address;
113 std::vector<estts::telemetry_object *> temp_telem;
114 bool associated_frame_found;
115 for (auto j : telem) {
116 if (command_id == j->commandID && command_address == j->address) {
117 // We have a match!
118 associated_frame_found = true;
119 temp_telem.push_back(j);
120 }
121 }
122 if (associated_frame_found) {
123 current->telem_obj = temp_telem;
124 current->response_code = validate_response_code(temp_telem[0]->response_code);
125 if (current->obj_callback != nullptr)
126 if (estts::ES_OK != current->obj_callback(current->telem_obj)) // Call the obj_callback with the telemetry object
127 return estts::ES_UNSUCCESSFUL;
128 // Todo this shoudn't return, it should schedule a new retry, or at least notify something that can retry
129
130 auto temp_completed = new completed;
131 temp_completed->serial_number = current->serial_number;
132 temp_completed->response_code = current->response_code;
133 completed_cache.push_back(temp_completed);
134 } else {
135 SPDLOG_WARN("Didn't receive a telemetry response for command with SN {}", current->serial_number);
136 current->response_code = estts::ES_UNSUCCESSFUL;
137 }
138 delete current;
139 dispatched.pop_back();
140 }
141
142 return estts::ES_OK;
143}
