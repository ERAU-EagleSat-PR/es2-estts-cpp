48estts::Status serial_handler::initialize_serial_port() {
49 boost::system::error_code error;
50
51 if (serial.is_open())
52 serial.close();
53
54 serial.open(port, error);
55 if (error) {
56 SPDLOG_ERROR("Failed to open serial port {} - {}", port, error.message());
57 return ES_UNSUCCESSFUL;
58 }
59
60 struct termios tty{};
61 if (tcgetattr(serial.lowest_layer().native_handle(), &tty) != 0) {
62 SPDLOG_ERROR("Error %i from tcgetattr: %s\n", errno, strerror(errno));
63 SPDLOG_INFO("Did you mean to use TI Dev Mode? See README.md");
64 return ES_UNSUCCESSFUL;
65 }
66
67 // Initialize Terminos structure
68 tty.c_lflag &= ~ICANON; // Disable UNIX Canonical mode (\n != terminator)
69 tty.c_lflag &= ~ECHO; // Disable echo
70 tty.c_lflag &= ~ECHOE; // Disable erasure
71 tty.c_lflag &= ~ECHONL; // Disable new-line echo
72 tty.c_lflag &= ~ISIG; // Disable interpretation of INTR, QUIT and SUSP
73 // tty.c_iflag &= ~(IXON | IXOFF | IXANY); // Disable software flow control
74 // tty.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR | ICRNL); // Disable any special handling of received bytes
75
76 // Save tty settings, also check for error
77 if (tcsetattr(serial.lowest_layer().native_handle(), TCSANOW, &tty) != 0) {
78 SPDLOG_ERROR("Error {} from tcsetattr: {}", errno, strerror(errno));
79 return ES_UNSUCCESSFUL;
80 }
81
82 try {
83 serial.set_option(boost::asio::serial_port_base::baud_rate(baud));
84 serial.set_option(boost::asio::serial_port_base::character_size(8));
85 serial.set_option(boost::asio::serial_port_base::flow_control(boost::asio::serial_port_base::flow_control::none));
86 serial.set_option(boost::asio::serial_port_base::parity(boost::asio::serial_port_base::parity::none));
87 serial.set_option(boost::asio::serial_port_base::stop_bits(boost::asio::serial_port_base::stop_bits::one));
88 } catch (boost::system::system_error::exception &e) {
89 SPDLOG_ERROR("Failed to configure serial port with baud {} - {}", baud, e.what());
90 return ES_UNSUCCESSFUL;
91 }
92
93 return estts::ES_OK;
94}

221 return [estts_callback, this] (const boost::system::error_code& error, std::size_t bytes_transferred) {

281Status serial_handler::find_serial_port() {
282 std::string serial_dir = "/dev/serial/by-id";
283 std::stringstream temp_sym;
284 DIR * d = opendir(serial_dir.c_str()); // open the path
285 if (d == nullptr) return estts::ES_UNSUCCESSFUL; // if was not able to open path, return
286 struct dirent * dir;
287 while ((dir = readdir(d)) != nullptr) {
288 if (dir->d_type == DT_LNK && strcmp(dir->d_name,".")!=0 && strcmp(dir->d_name,"..")!=0 ) {
289 // Basically if we get here whatever is inside by-id is a symlink pointing to the /dev
290 // location we want to connect to. We need to get the absolute path.
291 temp_sym << serial_dir << "/" << dir->d_name;
292 SPDLOG_INFO("Found serial device mounted at {}", temp_sym.str());
293 break;
294 }
295 }
296 if (temp_sym.str().empty()) {
297 SPDLOG_ERROR("Didn't find serial port to attach to. Ensure that device is plugged in and mounted.");
298 return ES_NOTFOUND;
299 }
300 this->port = temp_sym.str();
301 closedir(d);
302 return ES_OK;
303}
