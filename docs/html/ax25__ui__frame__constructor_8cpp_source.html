100 frameStream << getFlag();

104 frameStream << getDestAddr();
105 frameStream << getSSID0();
106 // Source address (hex encoded)
107 frameStream << getSrcAddr();
108 frameStream << getSSID1();
109 frameStream << getControl();
110 frameStream << getPID();
111 std::string info_field = getInfoField();
112 frameStream << info_field;
113 frameStream << calculate_crc16_ccit(info_field);
114 frameStream << getFlag();
115
116 // Postamble
117 for (int i = 0; i < 3; i++)
118 frameStream << getFlag();
119
120 SPDLOG_TRACE("Built AX.25 frame with value {}", frameStream.str());
121
122 std::string encoded_frame = this->encode_ax25_frame(frameStream.str());

124 SPDLOG_TRACE("Encoded AX.25 frame to {}", encoded_frame);
