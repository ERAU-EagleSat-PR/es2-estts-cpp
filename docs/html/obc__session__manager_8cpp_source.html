83void obc_session_manager::dispatch() {
84 using namespace std::this_thread; // sleep_for, sleep_until
85 using namespace std::chrono; // nanoseconds, system_clock, seconds
86 for (;;) {
87start:
88 if (!waiting.empty()) {
89 SPDLOG_TRACE("{} commands in queue", waiting.size());
90 if (!ti->obc_session_active) {
91 // Request a new communication session from EagleSat II
92 if (ES_OK != this->ti->request_obc_session()) {
93 SPDLOG_ERROR("Failed to request new session.");
94 goto start; // todo This should probably have a more elegant solution..
95 }
96 }
97
98 // After execute is called, the session is in progress. Set this state before, so that abstracted objects
99 // stay up to date.
100 SPDLOG_TRACE("Session status: {}", ti->obc_session_active);
101 if (ES_OK != this->execute(waiting.front())) {
102 SPDLOG_WARN("Failed to execute command with serial number {}", waiting.front()->serial_number);
103 } else {
104 SPDLOG_INFO("Command executed successfully");
105 }
106 waiting.pop_front();
107
108 if (waiting.empty() && ti->obc_session_active) {
109 if (ES_OK != this->ti->end_obc_session(ax25::END_SESSION_FRAME)) {
110 SPDLOG_WARN("Failed to end session rip");
111 }
112 SPDLOG_INFO("Waiting for more commands");
113 }
114
115 } else {
116
117 // Handle stream
118 auto stream = ti->nonblock_receive();
119 if (!stream.empty() && this->telem_callback != nullptr)
120 telem_callback(stream);
121 }
122 if (!ti->satellite_in_range) {
123 SPDLOG_DEBUG("OBC Dispatch worker - Detected satellite is outside range, exiting.");
124 return;
125 }
126 }
127}
