21std::string hex_to_ascii(const std::string& hex) {
22 // initialize the ASCII code string as empty.
23 std::string ascii = "";
24 for (size_t i = 0; i < hex.length(); i += 2)
25 {
26 // extract two characters from hex string
27 std::string part = hex.substr(i, 2);
28
29 // change it into base 16 and
30 // typecast as the character
31 char ch = stoul(part, nullptr, 16);
32
33 // add this char to final ASCII string
34 ascii += ch;

43std::string generate_serial_number() {
44 auto len = 16;
45 static const char alphanum[] =
46 "0123456789"
47 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
48 "abcdefghijklmnopqrstuvwxyz";
49 std::string tmp_s;
50 tmp_s.reserve(len);
51 std::random_device dev;
52 std::mt19937 rng(dev());
53 std::uniform_int_distribution<std::mt19937::result_type> dist6(1,sizeof(alphanum));
54
55 for (int i = 0; i < len; ++i) {
56 tmp_s += alphanum[dist6(rng) % (sizeof(alphanum) - 1)];
57 }
58
59 return tmp_s;
60}

69 struct dirent * dir;
70 while ((dir = readdir(d)) != nullptr) {
71 ssd_dir.clear();
72 if (strcmp(dir->d_name, ".")!=0 && strcmp(dir->d_name, "..")!=0 && !path_found) {
73 std::stringstream temp_path;
74 temp_path << ssd_dir.str() << dir->d_name;
75 DIR * d1 = opendir(temp_path.str().c_str());
76 if (d1 != nullptr) {
77 struct dirent * dir1;
78 while ((dir1 = readdir(d1)) != nullptr) {
79 if (dir->d_type == DT_DIR && strcmp(dir1->d_name, estts::REMOVABLE_STORAGE_NAME)==0) {
80 ssd_dir << dir->d_name << "/" << dir1->d_name;
81 SPDLOG_INFO("Constructed path to removable storage device - {}", ssd_dir.str());
82 path_found = true;
83 }
84 }
85 }
86 closedir(d1);
87 }
