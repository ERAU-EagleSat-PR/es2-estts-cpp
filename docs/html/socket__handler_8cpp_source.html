50estts::Status socket_handler::configure_socket() {
51 serv_addr.sin_family = AF_INET;
52 serv_addr.sin_port = htons(port);
53
54 // Convert IPv4 and IPv6 addresses from text to binary form
55 if (inet_pton(AF_INET, address, &serv_addr.sin_addr) <= 0) {
56 spdlog::error("Invalid address / Address not supported");
57 return estts::ES_UNSUCCESSFUL;
58 }
59
60 SPDLOG_DEBUG("Attempting to connect to socket at address {}:{}", address, port);
61
62 if (connect(sock, (struct sockaddr *) &serv_addr, sizeof(serv_addr)) < 0) {
63 spdlog::error("Error {} from connect(): {}", errno, strerror(errno));
64 return estts::ES_UNSUCCESSFUL;
65 }
66 int flags = fcntl(sock, F_GETFL, 0);
67 if (0 != fcntl(sock, F_SETFL, flags & ~O_NONBLOCK))
68 return estts::ES_UNSUCCESSFUL;
69 SPDLOG_TRACE("Connection succeeded.");
70
71 return estts::ES_OK;
72}

104unsigned char *socket_handler::read_socket_uc() const {
105 // If socket isn't open, error out
106 if (sock < 0) {
107 return nullptr;
108 }
109 // Allocate heap space for receive buffer
110 // auto buf = new unsigned char[estts::ti_socket::TI_SOCKET_BUF_SZ];
111
112 // Use read system call to read data in sock to buf
113 auto n = read(sock, sync_buf, estts::ti_socket::TI_SOCKET_BUF_SZ);
114 if (n < 1) {
115 // Can't receive a negative number of bytes ;)
116 return nullptr;
117 }
118
119 print_read_trace_msg(sync_buf, n, endpoint);
120
121 // Add null terminator at the end of transmission for easier processing by parent class(s)
122 sync_buf[n] = '\0';
123 return sync_buf;
124}

131estts::Status socket_handler::write_socket_s(const std::string &data) const {
132 // If serial port isn't open, error out
133 if (sock < 0) return estts::ES_UNINITIALIZED;
134 // Cast string to const unsigned char *, then cast away const to pass
135 // to method that writes unsigned char
136 auto csc_data = const_cast<unsigned char *>(reinterpret_cast<const unsigned char *>(data.c_str()));
137 if (this->write_socket_uc(csc_data, (int) data.length()) < 0) return estts::ES_UNSUCCESSFUL;
138 return estts::ES_OK;
139}

168estts::Status socket_handler::init_socket_handle() {

171 spdlog::error("Failed to open socket.");
