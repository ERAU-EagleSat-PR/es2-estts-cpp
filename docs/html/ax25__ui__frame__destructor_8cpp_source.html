91estts::Status ax25_ui_frame_destructor::build_telemetry_objects() {
92 try {
93 for (;;) {
94 // First, find the start of the frame using the flag.
95 // Assuming that the frame transmission had no loss, this method should work.
96 unsigned long frame_start_flag;
97 auto raw_length = raw_frame.length();
98 if (raw_length < 1) {
99 spdlog::error("Trying to decode empty frame. Exiting");
100 return estts::ES_UNSUCCESSFUL;
101 }
102 for (int i = 0; i < raw_length; i++) {
103 if (raw_frame[i] == estts::ax25::AX25_FLAG[0] && raw_frame[i + 1] == estts::ax25::AX25_FLAG[1]) {
104 i++;
105 } else {
106 frame_start_flag = i;
107 break;
108 }
109 }
110
111 // Now, we need to find the end of the frame. We can do this using the flag, and safely assume that if the
112 // trailing flag is missing, the frame is likely corrupted.
113 unsigned long frame_end;
114 for (int i = 0; i < raw_length; i++) {
115 if (i + 1 >= raw_length) {
116 SPDLOG_TRACE("No more frames found.");
117 return estts::ES_OK;
118 }
119 if (raw_frame[i] == estts::ax25::AX25_FLAG[0] && raw_frame[i + 1] == estts::ax25::AX25_FLAG[1]) {
120 if (estts::ax25::AX25_FLAG[1] != raw_frame[i - 1] && i > 0) {
121 frame_end = i;
122 break;
123 }
124 }
125 }
126
127 // Try decoding the frame.
128 auto frame = raw_frame.substr(frame_start_flag, frame_end - frame_start_flag);
129 if (estts::ES_OK == decode_frame(frame)) {
130 SPDLOG_TRACE("Trimming at index {} and looking for more frames", frame_end);
131 raw_frame.erase(0, frame_end);
132 } else {
133 // Then, find the start of the frame using the src/dest callsigns. This is a failsafe
134 // method in case the transmission had loss
135 std::stringstream callsign_and_ssid;
136 // Remember that we're expecting the satellite to be the source and ground to be the destination
137 callsign_and_ssid << ascii_to_hex(estts::ax25::AX25_SOURCE_ADDRESS) << estts::ax25::AX25_SSID0 <<
138 ascii_to_hex(estts::ax25::AX25_DESTINATION_ADDRESS) << estts::ax25::AX25_SSID1
139 << std::endl;
140 auto temp = raw_frame.find(callsign_and_ssid.str());
141 auto frame_start_callsign = (temp != std::string::npos) ? -1 : temp;
142 break;
143 }
144 }
145 }
146 catch (const std::exception &e) {
147 spdlog::error("Failed to find frame, found exception {}", e.what());
148 return estts::ES_UNSUCCESSFUL;
149 }
