55estts::Status posix_serial::initialize_serial_port() {
56 struct termios tty{};
57#ifndef __ESTTS_OS_LINUX__
58 if (tcgetattr(serial_port, &tty) != 0) {
59 SPDLOG_ERROR("Error %i from tcgetattr: %s\n", errno, strerror(errno));
60 SPDLOG_INFO("Did you mean to use TI Dev Mode? See README.md");
61 return ES_UNSUCCESSFUL;
62 }
63 // Initialize Terminos structure
64 tty.c_cflag &= ~PARENB; // Disable parity bit (IE clear parity bit)
65 tty.c_cflag &= ~CSTOPB; // 1 stop bit (IE clear stop field)
66 tty.c_cflag &= ~CSIZE; // Clear size bit
67 tty.c_cflag |= CS8; // 8 data bits
68 tty.c_cflag &= ~CRTSCTS; // Disable hardware flow control
69 tty.c_lflag &= ~ICANON; // Disable UNIX Canonical mode (\n != terminator)
70 tty.c_lflag &= ~ECHO; // Disable echo
71 tty.c_lflag &= ~ECHOE; // Disable erasure
72 tty.c_lflag &= ~ECHONL; // Disable new-line echo
73 tty.c_lflag &= ~ISIG; // Disable interpretation of INTR, QUIT and SUSP
74 tty.c_iflag &= ~(IXON | IXOFF | IXANY); // Disable software flow control
75 tty.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR |
76 ICRNL); // Disable any special handling of received bytes
77
78 // Set baud rate
79 cfsetispeed(&tty, baud);
80 cfsetospeed(&tty, baud);
81
82 // Set non-blocking
83 // todo make separate constructor determine if serial should read blocking/nonblocking
84 fcntl(serial_port, F_SETFL, FNDELAY);
85
86 // Save tty settings, also check for error
87 if (tcsetattr(serial_port, TCSANOW, &tty) != 0) {
88 SPDLOG_ERROR("Error {} from tcsetattr: {}", errno, strerror(errno));
89 return ES_UNSUCCESSFUL;
90 }
91#else
92 SPDLOG_INFO("Initializing serial handler with compatibility for Linux kernel");
93 if (tcgetattr(serial_port, &tty) != 0) {
94 spdlog::error("Error %i from tcgetattr: %s\n", errno, strerror(errno));
95 SPDLOG_INFO("Did you mean to use TI Dev Mode? See README.md");
96 return ES_UNSUCCESSFUL;
97 }
98
99 // Initialize Terminos structure
100 tty.c_cflag &= ~PARENB; // Disable parity bit (IE clear parity bit)
101 tty.c_cflag &= ~CSTOPB; // 1 stop bit (IE clear stop field)
102 tty.c_cflag &= ~CSIZE; // Clear size bit
103 tty.c_cflag |= CS8; // 8 data bits
104 tty.c_cflag &= ~CRTSCTS; // Disable hardware flow control
105 tty.c_cflag |= CREAD | CLOCAL; // Turn on READ and ignore control lines
106
107 tty.c_lflag &= ~ICANON; // Disable UNIX Canonical mode (\n != terminator)
108 tty.c_lflag &= ~ECHO; // Disable echo
109 tty.c_lflag &= ~ECHOE; // Disable erasure
110 tty.c_lflag &= ~ECHONL; // Disable new-line echo
111 tty.c_lflag &= ~ISIG; // Disable interpretation of INTR, QUIT and SUSP
112
113 tty.c_iflag &= ~(IXON | IXOFF | IXANY); // Disable software flow control
114 tty.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR |
115 ICRNL); // Disable any special handling of received bytes
116
117 tty.c_oflag &= ~OPOST;
118 tty.c_oflag &= ~ONLCR;
119
120 tty.c_cc[VTIME] = 100; // Set timeout of 10.0 seconds
121 tty.c_cc[VMIN] = 0;
122
123 // Set baud rate
124 cfsetispeed(&tty, B115200);
125 cfsetospeed(&tty, B115200);
126
127 cfmakeraw(&tty);
128
129 // Save tty settings, also check for error
130 if (tcsetattr(serial_port, TCSANOW, &tty) != 0) {
131 spdlog::error("Error {} from tcsetattr: {}", errno, strerror(errno));
132 return ES_UNSUCCESSFUL;
133 }
134 ioctl(serial_port, TIOCEXCL); // Disallow other processes from using this serial port
135#endif
136 SPDLOG_DEBUG("Successfully initialized port {} with fd {}", port, serial_port);
137 return ES_OK;

169ssize_t posix_serial::write_serial_uc(unsigned char *data, int size) const {
170 // If serial port isn't open, error out
171 if (serial_port < 0) {
172 return -1;
173 }
174 ssize_t written = write(serial_port, data, size);
175 if (written < 1) {
176 return -1;
177 }
178 if (data[size] == '\r')
179 SPDLOG_TRACE("Wrote '{}' (size={}) to {}", data, written, port);
180 else {
181 std::stringstream temp;
182 for (int i = 0; i < size; i ++) {
183 if (data[i] != '\r')
184 temp << data[i];
185 }
186 SPDLOG_TRACE("Wrote '{}' (size={}) to {}", temp.str(), written, port);
187 }
188 return written;
189}

199unsigned char *posix_serial::read_serial_uc() {
200 // Clear cache buf
201 cache.clear();
202 SPDLOG_TRACE("Reading serial - uc");
203
204 // If serial port isn't open, error out
205 if (serial_port < 0) {
206 SPDLOG_ERROR("Serial port not open! - serial_port = {}", serial_port);
207 return nullptr;
208 }
209 auto buf = new unsigned char[MAX_SERIAL_READ];
210 ssize_t n = read(serial_port, buf, MAX_SERIAL_READ);
211 if (n < 1) {
212 return nullptr;
213 }
214 if (buf[n] == '\r')
215 SPDLOG_TRACE("Read '{}' (size={}) from {}", buf, n, port);
216 else {
217 std::stringstream temp;
218 for (int i = 0; i < n; i ++) {
219 if (buf[i] != '\r')
220 temp << buf[i];
221 }
222 SPDLOG_TRACE("Read '{}' (size={}) from {}", temp.str(), n, port);
223 }
224 // Add null terminator at the end of transmission for easier processing by parent class(s)
225 buf[n] = '\0';
226 cache << buf;
227 return buf;
228}

235estts::Status posix_serial::write_serial_s(const std::string &data) const {
236 // If serial port isn't open, error out
237 if (serial_port < 0) return estts::ES_UNINITIALIZED;
238 // Cast string to const unsigned char *, then cast away const to pass
239 // to method that writes unsigned char
240 auto csc_data = const_cast<unsigned char *>(reinterpret_cast<const unsigned char *>(data.c_str()));
241 if (this->write_serial_uc(csc_data, (int) data.length()) < 0) return estts::ES_UNSUCCESSFUL;
242 return estts::ES_OK;
243}

249std::string posix_serial::read_serial_s() {
250 SPDLOG_TRACE("Reading serial - s");
251 // If serial port isn't open, error out
252 if (serial_port < 0) {
253 SPDLOG_ERROR("Serial port not open! - serial_port = {}", serial_port);
254 return "";
255 }
256 // Read serial data
257 auto read = this->read_serial_uc();
258 if (read == nullptr) {
259 return "";
260 }
261 // Type cast unsigned char (auto) to a char *
262 // Then call std::string constructor
263 std::string string_read(reinterpret_cast<char const *>(read));
264 delete read;
