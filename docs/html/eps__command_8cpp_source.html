8std::string eps_command::get_eps_vitals(const estts::dispatch_fct& dispatch, const std::function<estts::Status(estts::es2_telemetry::eps::vitals *)>& telem_callback) {
9 SPDLOG_INFO("Getting EagleSat II EPS Vitals");
10 auto command = new estts::command_object;
11
12 command->address = estts::es2_endpoint::ES_EPS;
13 command->commandID = estts::es2_commands::eps::EPS_GET_HEALTH;
14 command->method = estts::es2_commands::method::ES_READ;
15 command->sequence = 01;
16 command->timeStamp = 8456;
17
18
19
20 // This obj_callback is expected by the dispatcher/command handler, which will filter and construct telemetry objects
21 // and pass them to this function. Once this process is complete, this function should know how to decode
22 // the telemetry object into the expected structure (in this case, an EPS Vitals structure), and pass the vitals to
23 // the obj_callback function passed as an argument. This process keeps logic specific to EPS in the EPS
24 // command function, and changes the execution of the function to the command handler, where it should be. However, from
25 // the perspective of this EPS get vitals function, the specific time when the obj_callback is called doesn't matter in any way,
26 // because now the responsibility of handling the telemetry is pushed to the future when it's actually available.
27 auto eps_telem_decomposition_callback = [telem_callback] (const std::vector<estts::telemetry_object *>& telem) -> estts::Status {
28 if (telem.empty()) {
29 return estts::ES_UNINITIALIZED;
30 }
31 // TODO do something with the telem_obj vector passed to this function
32 auto vitals = new estts::es2_telemetry::eps::vitals;
33
34 vitals->battery_voltage = 9.0;
35 vitals->brownouts = 0;
36 vitals->charge_time_mins = 24;
37
38 spdlog::info("EPS Vitals: battery voltage: {} - brownouts: {} - charge time (min) {}", vitals->battery_voltage, vitals->brownouts, vitals->charge_time_mins);
39
40 telem_callback(vitals);
41 };
42
43 // We're expecting the dispatcher to call the telemetry decomposition function. As seen above, the telemetry
44 // decomposition knows how to handle a telemetry object of the EPS vitals type (which is matched by the address/command
45 // in the command configuration), and calls the obj_callback function passed in by the calling body. Note that when this function
46 // returns, it is expected that the entire telemetry collection and storage process is handled by the callbacks.
47 // This means that at no point does this function need to run again, because the context is implied by the lambdas.
48
49 // Note that our obj_callback model allows this function to return with no repercussions, and using the unique command
50 // serial number, we can fetch the command status at any time.
51 return dispatch(command, eps_telem_decomposition_callback);
52}
